Example with multiple list-attributes passed to DataGrid
:::note

It is only possible to pass multiple list fields to DataGrid if they are one-dimensional.

:::

```json title="cars.recipe.json" 
{
  "name": "cars",
  "type": "CORE:UiRecipe",
  "description": "List of cars",
  "plugin": "@development-framework/dm-core-plugins/data_grid",
  "config": {
    "type": "PLUGINS:dm-core-plugins/data_grid/DataGridPluginConfig",
    "fieldNames": [
      "red",
      "blue",
      "green"
    ],
    "columnLabels": [
      "NAME",
      "HEX",
      "RGB"
    ],
    "showRows": false
  }
} 
```
```json title="ColorList.blueprint.json" 
{
  "name": "ColorList",
  "type": "dmss://system/SIMOS/Blueprint",
  "attributes": [
    {
      "name": "type",
      "type": "dmss://system/SIMOS/BlueprintAttribute",
      "attributeType": "string"
    },
    {
      "name": "red",
      "type": "dmss://system/SIMOS/BlueprintAttribute",
      "attributeType": "string",
      "dimensions": "*"
    },
    {
      "name": "blue",
      "type": "dmss://system/SIMOS/BlueprintAttribute",
      "attributeType": "string",
      "dimensions": "*"
    },
    {
      "name": "green",
      "type": "dmss://system/SIMOS/BlueprintAttribute",
      "attributeType": "string",
      "dimensions": "*"
    }
  ]
} 
```

```json title="colors.entity.json" 
{
  "type": "./ColorList",
  "red": [
    "red",
    "#FF0000",
    "rgb(255, 0, 0)"
  ],
  "blue": [
    "blue",
    "#0000FF",

}
```
